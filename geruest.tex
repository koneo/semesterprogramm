\newcommand \neueSeite[4] {
	\newpage	
	\pagestyle{#1}							% Kopfzeile
	\seitenTitel{#3}						% Seitentitel
	\ifthenelse{#2}{\begin{center}}{}		% falls \zentiert,...
		\fontsize{\dfs}{\dls} \selectfont	% ... sonst
		#4									% Inhalt
		\ifthenelse{#2}{\end{center}}{}		% falls \zentriert,...
}
